void PrintSinglePiramid(int height)//одинарна піраміда
{
	for (int i = 0; i < height; i++)
	{
		for (int j = 0; j <= height - i; j++)//цикл for
		{
			cout << " ";
		}
		for (int k = 0; k <= i; k++)
		{
			cout << "#";
		}
		cout << endl;
	}
}
